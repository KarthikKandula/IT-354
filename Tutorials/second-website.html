Tutorial over Functions, Optional Arguments, Rest Paramenters & Scope

Functions

A JavaScript Function is a block of code or a set of statements that performs a particular task

An example JavaScript function can be seen below


        function exFunction(a, b) { // Function declaration
          return a + b; // returns addition of two parameters
        }
      
        var a = exFunction(2,3); // function call
        a // prints 5

Functions are mainly used to put some commonly used code together rather than writing the same code again and again

Function Syntax

A JavaScript function is defined with the 'function' keyword, followed by a name and a parentheses ()


          function name(parameter1, parameter2, parameter3) {
            // code to be executed
            return 'variable or value';
          }

return is used in some situations when we want to return some value from the function. return & parameters is optional

Function Expression

A JavaScript function can also be defined using an expression. A function expression can be stored in a variable


          // Function Expression
          var b = function() { return "This is a function expression!"; }
          console.log(b()) // prints "This is a function expression!"

Function name in a Function expression is optional. We refer the function through the variable

Named Functions

A named function is a a kind of function expression with the function name


          var b = function namedFunction() { return "This is a named function"; } // named function declaration
          console.log(b()) // prints "This is a named function"
          namedFunction(); // throws error

Named Functions cannot be called through the function name, it throws an error

Arrow Functions

An arrow function is defined using paranthesis containing list of parameters, followed by an arrow => and function body


          // Arrow function
          const arrowFunction = (number) => { // Arrow Function declaration
            if (number < 0) {
              console.log(-number)
            } else {
            console.log(number)
            }
          }
          arrowFunction(-10); // => prints '-10'
          arrowFunction(5);   // => prints '5'

Nested Functions

When a JavaScript function is declared inside a JavaScript function they are called Nested Functions. Below is an example of JavaScript Nested Functions. 


          // Nested Functions
          function fun1(a) {
            function fun2(b) {
              return a + b;
            }
            return fun2;
          }
          var c = fun1(4)(3);
          console.log(c)

Outer function cannot access variables inside inner function, but inner function can access variables of the outer function

Optional Arguments

If we don't supply enough values for the parameters to a function, JavaScript recognizes the missing arguments as 'undefined'. If we supply more arguments than required, JavaScript ignores all the extra arguments


          function optArg(a, b, c, d) { // Normal function declaratin with 4 arguments
            console.log(a) // prints '1'
            console.log(b) // prints '2'
            console.log(c) // prints 'undefined'
            console.log(d) // prints 'undefined'
          }
          
          optArg(1,2); // function call with only two parameters

Rest Parameters

Rest parameters in JavaScript allows to represent more than one number of arguments as an array


        // Rest parameters
        function add(...params) { // function which takes rest parameters
          return params;
        }

        console.log(add(1,2,3,4,5,6,7,8)) // prints [ 1, 2, 3, 4, 5, 6, 7, 8 ]
        console.log(add(1,2)) // prints [ 1, 2 ]
        console.log(add(1,2,3,4)) // prints [ 1, 2, 3, 4 ]

Another example of Rest Parameters


        // Rest parameters 2
        function restparams2(a , b, ...params) {
          return params;
        }

        console.log(restparams2(1,2,3,4,5,6,7,8)) // prints [3, 4, 5, 6, 7, 8 ]
        console.log(restparams2(1,2)) // prints []
        console.log(restparams2(1,2,3,4)) // prints [3, 4 ]

Scope

Scope can be defined as the range in which a variable is visible

In JavaScript there are two types of scope

  • Local Scope
  • Global Scope
Local Scope

Variables declared inside a function have local scope. i.e, variables that are not accessible to the whole document are called local variables


        // Scope
        // Local Variables
        function locVar() {
          var abc = "I am a Local Variable!"; // abc has local scope i.e, accessible only inside the function
          console.log(abc);
        }
        
        console.log('\n',"Scope")
        console.log("Local Variables", '\n')
        
        locVar();
        console.log(abc); // abc cannot be accessed here, throws an error
Global Scope

Variables declared outside a function have global scope. They are called Global Variables. These variables can be accessed all over the document.


        // Global Variables
        function globVar() {
          var abc = "I am a Local Variable!";
          console.log(abc);
          console.log(gloVar) // variable gloVar can be accessed inside the function because it is a Global Variable and it has global scope
        }
        
        console.log("Global Variables", '\n')
        
        var gloVar = "I am a Global Variable!"
        globVar();
